## Your First Component

Let's build a simple account balance card to demonstrate Bank Kit UI's capabilities.

### 1. Create the Component

Create a new file `components/account-balance.tsx`:

```tsx title="components/account-balance.tsx"
import { Card, CardContent, CardDescription, CardHeader, CardTitle } from "@bank-kit/ui/components/card";
import { Button } from "@bank-kit/ui/components/button";
import { Badge } from "@bank-kit/ui/components/badge";

interface AccountBalanceProps {
	accountName: string;
	balance: number;
	accountType: "checking" | "savings" | "credit";
	lastTransaction: string;
}

export function AccountBalance({ 
	accountName, 
	balance, 
	accountType, 
	lastTransaction 
}: AccountBalanceProps) {
	const formatCurrency = (amount: number) => {
		return new Intl.NumberFormat("en-US", {
			style: "currency",
			currency: "USD",
		}).format(amount);
	};

	const getAccountTypeColor = (type: string) => {
		switch (type) {
			case "checking":
				return "default";
			case "savings":
				return "secondary";
			case "credit":
				return "destructive";
			default:
				return "default";
		}
	};

	return (
		<Card className="w-full max-w-md">
			<CardHeader className="flex flex-row items-center justify-between space-y-0 pb-2">
				<CardTitle className="text-sm font-medium">
					{accountName}
				</CardTitle>
				<Badge variant={getAccountTypeColor(accountType)}>
					{accountType.charAt(0).toUpperCase() + accountType.slice(1)}
				</Badge>
			</CardHeader>
			<CardContent>
				<div className="text-2xl font-bold">
					{formatCurrency(balance)}
				</div>
				<CardDescription className="mt-2">
					Last transaction: {lastTransaction}
				</CardDescription>
				<div className="flex gap-2 mt-4">
					<Button size="sm" className="flex-1">
						Transfer
					</Button>
					<Button size="sm" variant="outline" className="flex-1">
						Details
					</Button>
				</div>
			</CardContent>
		</Card>
	);
}
```

### 2. Use the Component

Now use your new component in your application:

```tsx title="app/page.tsx"
import { AccountBalance } from "@/components/account-balance";

export default function Home() {
	return (
		<div className="container mx-auto p-6">
			<h1 className="text-3xl font-bold mb-8">Dashboard</h1>
			
			<div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-3 gap-6">
				<AccountBalance
					accountName="Primary Checking"

				/>
				<AccountBalance
					accountName="High Yield Savings"

					accountName="Credit Card"
					balance={-1205.43}
					accountType="credit"
					lastTransaction="1 hour ago"
				/>
			</div>
		</div>
	);
}
```

## Adding More Components

Let's enhance our dashboard with a transaction list:

### Transaction History Component

Create `components/transaction-list.tsx`:

```tsx title="components/transaction-list.tsx"
import { Card, CardContent, CardDescription, CardHeader, CardTitle } from "@bank-kit/ui/components/card";
import { Badge } from "@bank-kit/ui/components/badge";
import { Avatar, AvatarFallback } from "@bank-kit/ui/components/avatar";

interface Transaction {
	id: string;
	description: string;
	amount: number;
	date: string;
	category: string;
	type: "credit" | "debit";
}

interface TransactionListProps {
	transactions: Transaction[];
}

export function TransactionList({ transactions }: TransactionListProps) {
	const formatCurrency = (amount: number) => {
		return new Intl.NumberFormat("en-US", {
			style: "currency",
			currency: "USD",
		}).format(Math.abs(amount));
	};

	const getCategoryIcon = (category: string) => {
		const icons: Record<string, string> = {
			"Food": "üçî",
			"Transportation": "üöó",
			"Shopping": "üõçÔ∏è",
			"Income": "üí∞",
			"Bills": "üìÑ",
			"Entertainment": "üé¨",
		};
		return icons[category] || "üí≥";
	};

	return (
		<Card>
			<CardHeader>
				<CardTitle>Recent Transactions</CardTitle>
				<CardDescription>
					Your latest account activity
				</CardDescription>
			</CardHeader>
			<CardContent>
				<div className="space-y-4">
					{transactions.map((transaction) => (
						<div
							key={transaction.id}
							className="flex items-center justify-between space-x-4"
						>
							<div className="flex items-center space-x-4">
								<Avatar className="h-9 w-9">
									<AvatarFallback>
										{getCategoryIcon(transaction.category)}
									</AvatarFallback>
								</Avatar>
								<div>
									<p className="text-sm font-medium leading-none">
										{transaction.description}
									</p>
									<p className="text-sm text-muted-foreground">
										{transaction.date}
									</p>
								</div>
							</div>
							<div className="flex items-center space-x-2">
								<Badge variant="outline">
									{transaction.category}
								</Badge>
								<div className={`text-sm font-medium ${
									transaction.type === "credit" 
										? "text-green-600" 
										: "text-red-600"
								}`}>
									{transaction.type === "credit" ? "+" : "-"}
									{formatCurrency(transaction.amount)}
								</div>
							</div>
						</div>
					))}
				</div>
			</CardContent>
		</Card>
	);
}
```

## Key Features Demonstrated

This quick start showcases several important Bank Kit UI features:

1. **Type Safety**: Full TypeScript support with proper interfaces
2. **Financial Formatting**: Built-in currency formatting utilities
3. **Responsive Design**: Components that work across all devices
4. **Accessibility**: ARIA attributes and keyboard navigation built-in
5. **Consistent Styling**: Cohesive design system across all components
6. **Banking Context**: Components designed specifically for financial applications

## Best Practices

### Security

- Always validate and sanitize financial data
- Use HTTPS in production
- Implement proper authentication and authorization
- Follow PCI compliance guidelines for payment data

### Performance

- Use code splitting to reduce bundle size
- Implement lazy loading for large datasets
- Optimize images and icons
- Monitor Core Web Vitals

### Accessibility

- Test with screen readers
- Ensure keyboard navigation works
- Use semantic HTML elements
- Provide alt text for financial charts and graphs

### Error Handling

- Validate all user inputs
- Provide clear error messages
- Implement proper loading states
- Handle network failures gracefully

## Troubleshooting

Common issues and solutions:

**Styles not loading**: Ensure you've imported the global CSS file
**TypeScript errors**: Check your path mapping in tsconfig.json
**Components not found**: Verify your import paths match your project structure
**Responsive issues**: Check that you're using the responsive utility classes correctly

For more help, visit our [Troubleshooting Guide](/docs/getting-started/troubleshooting) or [GitHub Issues](https://github.com/bank-kit/ui/issues).
